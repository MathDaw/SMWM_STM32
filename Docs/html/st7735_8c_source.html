402 void st7735_DrawBitmap(uint16_t Xpos, uint16_t Ypos, uint8_t *pbmp)
403 {
404  uint32_t index = 0, size = 0;
405 
406  /* Read bitmap size */
407  size = *(volatile uint16_t *) (pbmp + 2);
408  size |= (*(volatile uint16_t *) (pbmp + 4)) << 16;
409  /* Get bitmap data address offset */
410  index = *(volatile uint16_t *) (pbmp + 10);
411  index |= (*(volatile uint16_t *) (pbmp + 12)) << 16;
412  size = (size - index)/2;
413  pbmp += index;
414 
415  /* Set GRAM write direction and BGR = 0 */
416  /* Memory access control: MY = 0, MX = 1, MV = 0, ML = 0 */
417  st7735_WriteReg(LCD_REG_54, 0x40);
418 
419  /* Set Cursor */
420  st7735_SetCursor(Xpos, Ypos);
421 
422  LCD_IO_WriteMultipleData((uint8_t*)pbmp, size*2);
423 
424  /* Set GRAM write direction and BGR = 0 */
425  /* Memory access control: MY = 1, MX = 1, MV = 0, ML = 0 */
426  st7735_WriteReg(LCD_REG_54, 0xC0);
427 }
