439 void BSP_LCD_DrawLine(uint16_t x1, uint16_t y1, uint16_t x2, uint16_t y2)
440 {
441  int16_t deltax = 0, deltay = 0, x = 0, y = 0, xinc1 = 0, xinc2 = 0,
442  yinc1 = 0, yinc2 = 0, den = 0, num = 0, numadd = 0, numpixels = 0,
443  curpixel = 0;
444 
445  deltax = ABS(x2 - x1); /* The difference between the x's */
446  deltay = ABS(y2 - y1); /* The difference between the y's */
447  x = x1; /* Start x off at the first pixel */
448  y = y1; /* Start y off at the first pixel */
449 
450  if (x2 >= x1) /* The x-values are increasing */
451  {
452  xinc1 = 1;
453  xinc2 = 1;
454  }
455  else /* The x-values are decreasing */
456  {
457  xinc1 = -1;
458  xinc2 = -1;
459  }
460 
461  if (y2 >= y1) /* The y-values are increasing */
462  {
463  yinc1 = 1;
464  yinc2 = 1;
465  }
466  else /* The y-values are decreasing */
467  {
468  yinc1 = -1;
469  yinc2 = -1;
470  }
471 
472  if (deltax >= deltay) /* There is at least one x-value for every y-value */
473  {
474  xinc1 = 0; /* Don't change the x when numerator >= denominator */
475  yinc2 = 0; /* Don't change the y for every iteration */
476  den = deltax;
477  num = deltax / 2;
478  numadd = deltay;
479  numpixels = deltax; /* There are more x-values than y-values */
480  }
481  else /* There is at least one y-value for every x-value */
482  {
483  xinc2 = 0; /* Don't change the x for every iteration */
484  yinc1 = 0; /* Don't change the y when numerator >= denominator */
485  den = deltay;
486  num = deltay / 2;
487  numadd = deltax;
488  numpixels = deltay; /* There are more y-values than x-values */
489  }
490 
491  for (curpixel = 0; curpixel <= numpixels; curpixel++)
492  {
493  BSP_LCD_DrawPixel(x, y, DrawProp.TextColor); /* Draw the current pixel */
494  num += numadd; /* Increase the numerator by the top of the fraction */
495  if (num >= den) /* Check if numerator >= denominator */
496  {
497  num -= den; /* Calculate the new numerator value */
498  x += xinc1; /* Change the x as appropriate */
499  y += yinc1; /* Change the y as appropriate */
500  }
501  x += xinc2; /* Change the x as appropriate */
502  y += yinc2; /* Change the y as appropriate */
503  }

877  width = DrawProp.pFont->Width;
878

880  *(uint16_t *) (bitmap + 2) = (uint16_t)(height*width*2+OFFSET_BITMAP);
881  *(uint16_t *) (bitmap + 4) = (uint16_t)((height*width*2+OFFSET_BITMAP)>>16);
882  *(uint16_t *) (bitmap + 10) = OFFSET_BITMAP;
883  *(uint16_t *) (bitmap + 18) = (uint16_t)(width);
884  *(uint16_t *) (bitmap + 20) = (uint16_t)((width)>>16);
885  *(uint16_t *) (bitmap + 22) = (uint16_t)(height);
886  *(uint16_t *) (bitmap + 24) = (uint16_t)((height)>>16);
